const nameInput = document.getElementById("nameInput");
    const scoreInput = document.getElementById("scoreInput");
    const addBtn = document.getElementById("addBtn");
    const bestBtn = document.getElementById("bestBtn");
    const scoreList = document.getElementById("scoreList");
    const result = document.getElementById("result");

    const students = [];

addBtn.addEventListener("click", () => {
    const name = nameInput.value.trim();
    const score = parseFloat(scoreInput.value);

    if (!name || isNaN(score) || score < 0 || score > 100) {
        alert("Введите корректные данные!");
        return;
    }

    students.push({ name, score });

    const li = document.createElement("li");
    li.textContent = `${name} — ${score}`;
    scoreList.appendChild(li);

    nameInput.value = "";
    scoreInput.value = "";
    nameInput.focus();
});

bestBtn.addEventListener("click", () => {
    if (students.length === 0) {
        result.textContent = "Нет оценок!";
        result.style.color = "red";
        return;
    }

    // Находим максимальную оценку
    const maxScore = Math.max(...students.map(s => s.score));
    const bestStudents = students.filter(s => s.score === maxScore).map(s => s.name);

    result.textContent = `Лучшая оценка: ${maxScore} — ${bestStudents.join(", ")}`;
    result.style.color = "darkblue";
});
