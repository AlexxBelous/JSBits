const input = document.getElementById("priceInput");
    const button = document.getElementById("roundBtn");
    const result = document.getElementById("result");

    button.addEventListener("click", () => {
    let price = parseFloat(input.value);

    if (isNaN(price)) {
        result.textContent = "Введите корректное число!";
        result.style.color = "red";
    } else {
        let rounded = price.toFixed(2);
        result.textContent = `Округленная цена: ${rounded}`;
        result.style.color = "darkblue";
    }
});
  • Значение берётся из input через parseFloat.
  • Проверка isNaN не даёт ввести ерунду.
  • Используется .toFixed(2) для округления.
  • Результат появляется в div#result.
